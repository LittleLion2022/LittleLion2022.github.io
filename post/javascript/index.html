5.javascript中0、NaN、空字符串"“视为false,其它视为true

function power(base,exponent){
    if(exponent == undefined)
        exponent = 2;
    var result = 1;
    for(var count = 0;count < exponent;count++)
        result *= base;
    return result;
}

console.log(power(4));
